import pandas as pd
import sqlite3
%matplotlib inline
def run_query(q):
    with sqlite3.connect('chinook.db') as conn:
        return pd.read_sql(q, conn)
def run_command(c):
    with sqlite3.connect('chinook.db') as conn:
        conn.isolation_level = None
        conn.execute(c)
def show_tables():
    q = 'SELECT name, type FROM sqlite_master WHERE type IN ("table","view");'
    return run_query(q)

#NOW we are going to find the total number of customers, sales and average sales 
# per customer. We group countries with 1 customer into one catergory 'other'
run_command("""CREATE VIEW main AS
select c.country, SUM(i.total) total_sales, COUNT(DISTINCT(c.customer_id)) customer_count,
    SUM(i.total) / CAST(COUNT(DISTINCT(c.customer_id)) as float) avg_sale   
    from customer c 
    INNER JOIN invoice i on c.customer_id=i.customer_id GROUP BY 1
""")
